# 建立一个存放emacs配置的文件夹，用于存放各种emacs的配置，方便切换
# 同时支持多用户
mkdir /opt/emacs.d/
chmod  777 /opt/emacs.d/

# purcell's .emacs.d
git clone https://github.com/purcell/emacs.d.git purcell
ln -s /opt/emacs.d/purcell ~/.emacs.d
[ 卸载：rm -rf ~/.emacs.d  # not is ~/.emacs.d/ ]
